program num_primo?;
var
  int: x&, i&;
  string: res#;
  bool: esPrimo$;

begin
    read(x&);
    if (x& <= 1) then
       begin
            res# := "No es un numero primo";
       end
    else
       begin
            esPrimo$ := true;
            if (esPrimo$) then
              begin
                 res# := "Es un numero primo";
              end
            else
              begin
                 res# := "No es un numero primo";
              end
       end
    write(x&);
    write(res#);
end
